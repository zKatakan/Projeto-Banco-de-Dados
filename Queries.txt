create table professor(
    	nome varchar(255),
	prof_id varchar(20),
    	dept_id varchar(20),
	primary key (prof_id)
);

create table historico_professor(
	prof_id varchar(20),
	disc_id varchar(20),
	semestre int,
	ano int,
    	primary key (prof_id)
);

create table aluno(
	nome varchar(255),
	aluno_id varchar(20),
	curso_id varchar(20),
	primary key (aluno_id)
);

create table historico_aluno(
	aluno_id varchar(20),
	disc_id varchar(20),
	curso_id varchar(20),
	nota int,
	semestre int,
	ano int,
    	primary key (aluno_id)
);

create table curso (
   	nome varchar(255),
	curso_id varchar(20),
	dept_id varchar(20),
	primary key (curso_id)
);

create table matriz_curricular (
	disc_id varchar(20),
	curso_id varchar(20),
   	semestre int
);

create table disciplina (
	nome varchar(255),
	disc_id varchar(20),
	dept_id varchar(20),
	primary key (disc_id)
);

create table departamento (
	nome varchar(255),
	dept_id varchar(20),
	chefe_id varchar(20),
	primary key (dept_id)
);

create table tcc (
	nome varchar(255),
	tcc_id varchar(20),
	curso_id varchar(20),
	prof_id varchar(20),
	primary key (tcc_id)
);

create table grupo_tcc (
	tcc_id varchar(20),
	aluno_id varchar(255)
);
