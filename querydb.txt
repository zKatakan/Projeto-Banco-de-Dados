create table professor(
	prof_id varchar(20),
	nome	varchar(255),
	primary key (prof_id)
	);
create table historico_professor(
	prof_id varchar(20),
	disc_id varchar(20),
	semestre int,
	ano int
	);

create table aluno(
	aluno_id varchar(20),
	nome varchar(255),
	curso_id varchar(20),
	primary key (aluno_id)
	);
create table historico_aluno(
	aluno_id varchar(20),
	disc_id varchar(20),
	nota int,
	semestre int,
	ano int
	);
create table curso (
	curso_id varchar(20),
	nome varchar(255),
	dept_id varchar(20),
	primary key (curso_id)
	);
create table curso_disc (
	disc_id varchar(255),
	curso_id varchar(20)
	);
create table disciplina (
	disc_id varchar(20),
	nome varchar(255),
	dept_id varchar(20),
	primary key (disc_id)
	);
create table departamento (
	dept_id varchar(20),
	nome varchar(255),
	chefe_id varchar(20),
	primary key (dept_id)
	);
create table tcc (
	tcc_id varchar(20),
	nome varchar(255),
	prof_id varchar(20),
	curso_id varchar(20),
	ano int,
	semestre int,
	primary key (tcc_id)
	);
create table part_tcc (
	tcc_id varchar(20),
	aluno_id varchar(255)
	
	);
